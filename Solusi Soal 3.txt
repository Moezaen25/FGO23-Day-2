3. Notasi Algoritma Pseudocode

PROGRAM PERHITUNGAN_BESAR_DISKON
	DECLARE BELANJA, BAYAR, DISKON: number
BEGIN
	INPUT(BELANJA)
	IF BELANJA < 500.000:
		DISKON = 0
	  ELSE IF BELANJA >= 500.000:
		  DISKON = 5%*BELANJA
	    ELSE IF BELANJA > 1.000.000
		    DISKON = 10%*BELANJA
	    END IF
	  END IF
	END IF
	BAYAR = BELANJA-DISKON
	OUTPUT(DISKON, BAYAR)
END
